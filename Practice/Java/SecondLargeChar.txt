import java.util.*;

class SecondLargeChar {
	public static void main(String[] args) {
		int[] arr = { 12, 35, 1, 10, 34, 1 };
		Arrays.sort(arr);
		
		System.out.println("SecondLargeChar: "+SecondLarge(arr));
		
	}
	
	static int SecondLarge(int[] input){
	    int n = input.length;
	    for(int i=n-2;i>=0;i--){
	        if(input[i]!=input[n-1]){
	            return input[i];
	        }
	    }
	    
	    return -1;
	}
}
