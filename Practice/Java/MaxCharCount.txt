import java.util.*;

class MaxCharCount {
	public static void main(String[] args) {
		String s = "Try programiz testing interviewssss";
		s = s.replace(" ", "").toLowerCase();
		Map < Character, Integer > mp = new HashMap < > ();
		int maxCount = 0;

		for (int i = 0; i < s.length(); i++) {
			if (mp.containsKey(s.charAt(i))) {
				int occur = mp.get(s.charAt(i));
				occur++;
				mp.put(s.charAt(i), occur);
				if (maxCount < occur)
					maxCount = occur;
			} else {
				mp.put(s.charAt(i), 1);
			}
		}

		for (Map.Entry < Character, Integer > m: mp.entrySet()) {
			int temp2 = m.getValue();
			if (maxCount == temp2) {
				System.out.println("Maximum repeated character and count from given string");
				System.out.println("Word: " + m.getKey() + " & length: " + m.getValue());
			}
		}
	}
}
